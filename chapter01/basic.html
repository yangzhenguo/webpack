基本示例

index.html

<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <title>Webpack</title>
</head>
<body>
<div id="app"></div>
<script src="dist/bundle.js"></script>
</body>
</html>

Webpack配置文件webpack.config.js

const path = require('path');

module.exports = {
    entry: './src/main.js',
    output: {
        path: path.resolve(__dirname, 'dist'),
        filename: 'bundle.js'
    }
}

自定义CommonJS模块print.js

module.exports = function (text) {
    return document.querySelector('#app').innerHTML = 'Hello, ' + text;
}

引入CommonJS模块main.js

const print = require('./print');
print('Webpack');
